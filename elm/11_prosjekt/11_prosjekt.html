Introduksjon

Har du noe du har hatt lyst til å lage en stund? Nå er sjansen! Under er noen forslag for å komme i gang.

Idé 1: spill

Her er en måte å sjekke hva brukeren trykker på av knapper:

module Main exposing (..)

import Html exposing ( text )
import Keyboard

type Msg = Down Int
         | Up Int

model =
    { lastDown = 0
    , lastUp = 0
    }

main =
    Html.program
       { init = ( model, Cmd.none )
       , view = view
       , update = update
       , subscriptions = subscriptions
       }

update msg model =
    let
        newmodel =
            case msg of
                Down d -> { model | lastDown = d }
                Up u -> { model | lastUp = u}
    in
        ( newmodel, Cmd.none )

view model = text ( "Last down: " ++ toString model.lastDown
                    ++ ", last up: " ++ toString model.lastUp )

subscriptions model =
    Sub.batch
        [ Keyboard.downs Down
        , Keyboard.ups Up
        ]
